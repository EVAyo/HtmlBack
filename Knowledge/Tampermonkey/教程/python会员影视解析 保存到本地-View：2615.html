利用第三方接口 抓包m3u8真实地址
找到地址后 取出m3u8中的ts文件 遍历下载 下载到本地后 利用os库执行cmd命令 合并成一个视频 实现下载功能
练手写的，大概思路就这样，具体操作忘了。

遍历速度慢，有能力自己加多线程，加起来不难。

python代码：
  1. import requests
  2. import re
  3. import os#创建文件夹 删除文件夹等文件夹操作
  4. import sys#取运行目录


  7. def Getmu38(vieo_url):
  8.     url='接口自己抓'+vieo_url
  9.     print('\n正在解析中......')
  10.     res = requests.get(url)
  11.     res = res.json()
  12.     #print(res)
  13.     if res['code'] == '200':
  14.         m3u8_url = res['url']
  15.         #print(m3u8_url)
  16.         Get_ts(m3u8_url)
  17.         print(m3u8_url)
  18.         return m3u8_url
  19.     else:
  20.         print('解析失败，请检查网络或重新解析！')
  21. def Get_ts(m3u8_url):
  22.     res = requests.get(m3u8_url)
  23.     ts_list = re.compile(r'[a-zA-z]+://[^\s]*').findall(res.text)
  24.     path = sys.path[0]
  25.     print('当前运行目录：'+path)
  26.     if len(ts_list) ==0:
  27.         print('处理失败，请重新尝试！')
  28.     else:
  29.         os.makedirs(path + "\视频")
  30.         for i in range(len(ts_list)):
  31.             ts_url = ts_list[i]
  32.             # print(ts_url)
  33.             res_ts = requests.get(ts_url)
  34.             with open(path + '\视频\\' + str(i) + '.ts', 'wb') as f:
  35.                 f.write(res_ts.content)
  36.             print('共需处理' + str(len(ts_list)-1) + '个数据，已处理' + str(i) + '个数据！')
  37.         video_name = input('请输入此电影的名称：')
  38.         os.system('cd ' + path + '\视频\\' + '&& copy /b *.ts '+video_name +'.mp4')
